<script>

    // Official Real-Time Matchups for the 2026 Round of 32 (Starting June 28, 2026)

    const initialTeams = [

        ["South Africa", "Canada"], ["Brazil", "Japan"],

        ["Germany", "Paraguay"], ["Netherlands", "Morocco"],

        ["Ivory Coast", "Norway"], ["France", "Sweden"],

        ["Mexico", "Ecuador"], ["England", "DR Congo"],

        ["Belgium", "Senegal"], ["United States", "Bosnia"],

        ["Spain", "Austria"], ["Portugal", "Croatia"],

        ["Switzerland", "Algeria"], ["Australia", "Egypt"],

        ["Argentina", "Cabo Verde"], ["Colombia", "Ghana"]

    ];


    let currentProfile = 0;

    // Load state from URL parameters if available, else initialize blank grids for 4 players

    let globalStore = [{}, {}, {}, {}];


    // Read URL configurations if friend shares it

    const urlParams = new URLSearchParams(window.location.search);

    if(urlParams.has('d')) {

        try {

            globalStore = JSON.parse(atob(urlParams.get('d')));

        } catch(e) { console.error("Data parse failure", e); }

    }


    const roundNames = ["Round of 32", "Round of 16", "Quarter-finals", "Semi-finals", "Final", "Champion"];


    function buildBracket() {

        const container = document.getElementById('bracket');

        container.innerHTML = '';

        

        let currentRoundMatches = initialTeams.map((teams, index) => ({ id: `r1-m${index}`, teams }));

        let profileData = globalStore[currentProfile];


        for (let r = 0; r < 5; r++) {

            const roundDiv = document.createElement('div');

            roundDiv.className = 'round';

            roundDiv.innerHTML = `<div class="round-title">${roundNames[r]}</div>`;


            let nextRoundMatches = [];


            for (let m = 0; m < currentRoundMatches.length; m++) {

                const match = currentRoundMatches[m];

                const matchDiv = document.createElement('div');

                matchDiv.className = 'matchup';


                const savedWinner = profileData[match.id];

                

                match.teams.forEach(team => {

                    const slot = document.createElement('div');

                    slot.className = `team-slot ${savedWinner === team ? 'selected' : ''}`;

                    slot.innerText = team || 'TBD';

                    if(team) {

                        slot.onclick = () => selectWinner(r, m, match.id, team);

                    }

                    matchDiv.appendChild(slot);

                });


                roundDiv.appendChild(matchDiv);


                // Setup structure nodes for subsequent structural generation

                if (m % 2 === 0) {

                    let nextMatchId = `r${r+2}-m${Math.floor(m/2)}`;

                    let nextTeams = [profileData[match.id] || null];

                    let siblingMatch = currentRoundMatches[m+1];

                    if (siblingMatch) {

                        nextTeams.push(profileData[siblingMatch.id] || null);

                    }

                    nextRoundMatches.push({ id: nextMatchId, teams: nextTeams });

                }

            }

            container.appendChild(roundDiv);

            currentRoundMatches = nextRoundMatches;

        }


        // Champion Output Display

        const champRound = document.createElement('div');

        champRound.className = 'round';

        champRound.innerHTML = `<div class="round-title">Champion</div>`;

        const champMatch = currentRoundMatches[0];

        const champName = profileData[champMatch?.id] || '🏆 Winner';

        champRound.innerHTML += `<div class="matchup" style="border-color:gold; text-align:center; padding: 20px; font-weight:800; color:gold;">${champName}</div>`;

        container.appendChild(champRound);

    }


    function selectWinner(roundIdx, matchIdx, matchId, selectedTeam) {

        globalStore[currentProfile][matchId] = selectedTeam;

        // Invalidate downstream dependent paths if they flip an older choice

        invalidateDownstream(roundIdx + 2, matchIdx, selectedTeam);

        buildBracket();

    }


    function invalidateDownstream(startRound, prevMatchIdx, chosenTeam) {

        let currentMatchIdx = Math.floor(prevMatchIdx / 2);

        let activeSelection = chosenTeam;

        for (let r = startRound; r <= 5; r++) {

            let mid = `r${r}-m${currentMatchIdx}`;

            // If downstream choice doesn't equal what migrated up, clear it

            if(globalStore[currentProfile][mid] && globalStore[currentProfile][mid] !== activeSelection) {

                 // dynamic cascading reset rules can be layered here if desired

            }

            currentMatchIdx = Math.floor(currentMatchIdx / 2);

        }

    }


    function switchProfile(id) {

        currentProfile = id;

        document.querySelectorAll('.profile-btn').forEach((btn, idx) => {

            btn.classList.toggle('active', idx === id);

        });

        buildBracket();

    }


    function shareBracket() {

        const payload = btoa(JSON.stringify(globalStore));

        const shareUrl = `${window.location.origin}${window.location.pathname}?d=${payload}`;

        navigator.clipboard.writeText(shareUrl).then(() => {

            alert("🔗 Shareable application URL copied to clipboard! Send it to your 3 friends.");

        });

    }


    buildBracket();
